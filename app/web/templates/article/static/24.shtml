php5 连接oracle 10g（心血啊）
2014-12-19
/*
Oracle数据库服务器IP：192.168.0.254
Oracle数据库服务器端口：1521
Oracle数据库用户账号：test
Oracle数据库用户密码：test123
Oracle数据库服务器SID：oracle
*/
// oracle 连接字符串
$db = \"(DESCRIPTION=(ADDRESS_LIST = (ADDRESS = (PROTOCOL = TCP)(HOST = 192.168.0.254)(PORT = 1521)))(CONNECT_DATA=(SID=oracle)))\";
// 连接oracle
$conn = oci_connect(\'test\', \'test123\',$db,\'UTF8\');
// 返回sql执行标示
$stid = oci_parse($conn, \'select id,name  from pictest\');
// 执行sql语句
oci_execute($stid);
// 循环获取值
while (($row = oci_fetch_array($stid, OCI_BOTH))) {
    echo $row[0] . \" and \" . $row[\'id\']   . \" are the same<br>\\n\";
    echo $row[1] . \" and \" . $row[\'name\'] . \" are the same<br>\\n\";
}
// 释放资源
oci_free_statement($stid);
// 关闭连接
oci_close($conn);
